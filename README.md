### utils.execQuery(data, query)
```js
var data = {
    id: 1,
    name: "raccoon",
    type: "animal",
    size: {
        height: 20,
        weight: 100,
    },
    array: [1,2,3],
    array2: null
}
var query = {
    id: 1,
    name: {
        $in: ["raccoon", "abcdefg"],
        $nin: ["dog", "cat"],
        $regexp: /raccoon/
    },
    type: /^ani/, // regexp
    $and: [{
        size: {
            $gt: 1, // object keys length
            $lt: 3
        }
    }, {
        size: {
            $or: [{
                weight: {
                    $gt: 90, // number
                    $lt: 110
                }
            }, {
                height: {
                    $gte: 20, // number
                    $lte: 20
                }
            }]
        }
    }],
    array: {
        $gt: 2, // array length
        $lt: 4
    },
    array2: {
        $exists: false // null or undefined
    }
}

var res = utils.execQuery(data, query);
// true
```
